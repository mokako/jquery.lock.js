jQuery-lock.js

Documentation & DEMO

FULL

$(function(){
  $(".lock").lock({
    pattern:'n',
    match : 3,
    comment : "Please click table",
    limitCount : 2,
    tdWidth :100,
    tdHeight :100,
    cols : 4,
    rows : 4,
    lock : function(){
      userFunction();
    },
    unlock : function(){
      userFunction();
    },
    limit : function(){
      userFunction();
    }
  });
});
PATTERN

pattern : 'n' (number)

$(".lock").lock({
  pattern:'n'
});

pattern : 'a' (alphabet)

$(".lock").lock({
  pattern:'a'
});

pattern : 'r' (random)

$(".lock").lock({
  pattern:'r'
});
MATCH

match : (number)

$(".lock").lock({
  match : 2
});
COMMENT

comment : "character" or false 

$(".lock").lock({
  comment : "Press table"
});
LIMITCOUNT

limitCount : 2 (limitCout + match = limit)

$(".lock").lock({
 limitCount : 2,
 limit : function(){
  	$(this).fadeOut("slow");
  }
});
TABLE WIDTH + HEIGHT

tdWidth , tdHeight

$(".lock").lock({
  tdWidth:30,
  tdHeight:30
});
TABLE COLS + ROWS

cols , rows

$(".lock").lock({
  cols:7,
  rows:2
});
LOCK ,UNLOCL ,LIMIT (CALLBACK)

lock : function(){} , unlock : function(){} , limit : function(){}

$(".lock").lock({
  lock : function(){
    $(this)
      .css("background-color","#0074CC");	
  },
  unlock : function(){
    $(this)
      .css("background-color","#FFFFFF");
  },
  limit : function(){
    $(this).fadeOut("slow");
  }
});
